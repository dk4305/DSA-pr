Pr.8 primary key and foreign key

-- Create Author table 
CREATE TABLE Author ( 
author_id INT PRIMARY KEY, 
name VARCHAR(100) NOT NULL, 
birth_year INT 
); -- Create Book table 
CREATE TABLE Book ( 
book_id INT PRIMARY KEY, 
title VARCHAR(100) NOT NULL, 
author_id INT, 
published_year INT, 
FOREIGN KEY (author_id) REFERENCES Author(author_id) 
); -- Create Borrower table 
CREATE TABLE Borrower ( 
borrower_id INT PRIMARY KEY, 
name VARCHAR(100) NOT NULL, 
contact_info VARCHAR(255), 
membership_type VARCHAR(50) 
); -- Alter tables: Add and Modify 
ALTER TABLE Book ADD genre VARCHAR(50); 
ALTER TABLE Borrower ADD membership_type VARCHAR(50); 
ALTER TABLE Borrower MODIFY contact_info VARCHAR(500); -- Dropping tables 
DROP TABLE Borrower; 
DROP TABLE Book; 
DROP TABLE Author;
