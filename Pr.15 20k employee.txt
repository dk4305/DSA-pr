15. Database Schema for a Employee-pay scenario employee(emp_id : integer, emp_name:
string) department(dept_id: integer, dept_name:string) paydetails(emp_id : integer, dept_id:
integer, salary: integer, DOJ: date) payroll(emp_id : integer, pay_date: date) Write a trigger
which will restrict insertion of new employee record having salary less than Rs. 20000.
-- Create Employee Table
CREATE TABLE employee (
 emp_id INT PRIMARY KEY,
 emp_name VARCHAR(100)
);
-- Create Department Table
CREATE TABLE department (
 dept_id INT PRIMARY KEY,
 dept_name VARCHAR(100)
);
-- Create Paydetails Table
CREATE TABLE paydetails (
 emp_id INT,
 dept_id INT,
 salary INT,
 DOJ DATE,
 PRIMARY KEY (emp_id, dept_id),
 FOREIGN KEY (emp_id) REFERENCES employee(emp_id),
 FOREIGN KEY (dept_id) REFERENCES department(dept_id)
);
-- Create Payroll Table
CREATE TABLE payroll (
 emp_id INT,
 pay_date DATE,
 PRIMARY KEY (emp_id, pay_date),
 FOREIGN KEY (emp_id) REFERENCES employee(emp_id)
);
DELIMITER $$
CREATE TRIGGER RestrictSalaryInsertion
BEFORE INSERT ON paydetails
FOR EACH ROW
BEGIN
 -- Check if the salary is less than Rs. 20000
 IF NEW.salary < 20000 THEN
 SIGNAL SQLSTATE '45000'
 SET MESSAGE_TEXT = 'Salary cannot be less than Rs. 20000';
 END IF;
END $$
DELIMITER ;
-- Insert a valid employee record with salary >= 20000
INSERT INTO paydetails (emp_id, dept_id, salary, DOJ)
VALUES (1, 101, 25000, '2025-04-01');
-- Insert an invalid employee record with salary < 20000
INSERT INTO paydetails (emp_id, dept_id, salary, DOJ)
VALUES (2, 102, 15000, '2025-04-02');
