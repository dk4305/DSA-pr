Pr.4  Video Library scenario Customer

CREATE TABLE Customer ( 
cust_no INT PRIMARY KEY, 
cust_name VARCHAR(100) NOT NULL 
); 
CREATE TABLE Membership ( 
mem_no INT PRIMARY KEY, 
cust_no INT, 
FOREIGN KEY (cust_no) REFERENCES Customer(cust_no) 
); 
CREATE TABLE Cassette ( 
cass_no INT PRIMARY KEY, 
cass_name VARCHAR(100) NOT NULL, 
Language VARCHAR(50) 
); 
CREATE TABLE Iss_rec ( 
iss_no INT PRIMARY KEY, 
iss_date DATE NOT NULL, 
mem_no INT, 
cass_no INT, 
FOREIGN KEY (mem_no) REFERENCES Membership(mem_no), 
FOREIGN KEY (cass_no) REFERENCES Cassette(cass_no) 
);

List all the customer names with their membership numbers 
SELECT c.cust_name, m.mem_no 
FROM Customer c 
JOIN Membership m ON c.cust_no = m.cust_no; 
List all the issues for the current date with customer names and cassette names 
SELECT c.cust_name, cs.cass_name, ir.iss_no, ir.iss_date 
FROM Iss_rec ir 
JOIN Membership m ON ir.mem_no = m.mem_no 
JOIN Customer c ON m.cust_no = c.cust_no 
JOIN Cassette cs ON ir.cass_no = cs.cass_no 
WHERE ir.iss_date = CURDATE(); 
Details of customer who borrowed the cassette titled “The Legend” 
SELECT c.* 
FROM Iss_rec ir 
JOIN Membership m ON ir.mem_no = m.mem_no 
JOIN Customer c ON m.cust_no = c.cust_no 
JOIN Cassette cs ON ir.cass_no = cs.cass_no 
WHERE cs.cass_name = 'The Legend'; 
Count of how many cassettes each customer has borrowed 
SELECT c.cust_name, COUNT(ir.cass_no) AS total_borrowed 
FROM Iss_rec ir 
JOIN Membership m ON ir.mem_no = m.mem_no 
JOIN Customer c ON m.cust_no = c.cust_no 
GROUP BY c.cust_name; 
List of cassettes taken by the customer with mem_no = 5 
SELECT cs.* 
FROM Iss_rec ir 
JOIN Cassette cs ON ir.cass_no = cs.cass_no 
WHERE ir.mem_no = 105; -- assuming 105 is the desired membership number 
List the cassettes issued today 
SELECT cs.* 
FROM Iss_rec ir 
JOIN Cassette cs ON ir.cass_no = cs.cass_no 
WHERE ir.iss_date = CURDATE(); 
Create a view: iss_no, iss_date, cust_name, cass_name 
CREATE VIEW IssuedCassetteView AS 
SELECT  
ir.iss_no,  
ir.iss_date,  
c.cust_name,  
cs.cass_name 
FROM Iss_rec ir 
JOIN Membership m ON ir.mem_no = m.mem_no 
JOIN Customer c ON m.cust_no = c.cust_no 
JOIN Cassette cs ON ir.cass_no = cs.cass_no; 
Create a view listing issues date-wise for the last one week 
CREATE VIEW WeeklyCassetteIssues AS 
SELECT  
ir.iss_date,  
COUNT(*) AS total_issues 
FROM Iss_rec ir 
WHERE ir.iss_date >= CURDATE() - INTERVAL 6 DAY
GROUP BY ir.iss_date 
ORDER BY ir.iss_date;
