14. Consider following database: Student(Stud_no : integer, Stud_name: string)
Membership(Mem_no: integer, Stud_no: integer) Book(book_no: integer, book_name:string,
author: string) Iss_rec(iss_no:integer, iss_date: date, Mem_no: integer, book_no: integer) Write
a trigger to update quantity of book if that book is issued to student.
-- Create Student Table
CREATE TABLE Student (
 Stud_no INT PRIMARY KEY,
 Stud_name VARCHAR(100)
);
-- Create Membership Table
CREATE TABLE Membership (
 Mem_no INT PRIMARY KEY,
 Stud_no INT,
 FOREIGN KEY (Stud_no) REFERENCES Student(Stud_no)
);
-- Create Book Table (with quantity column)
CREATE TABLE Book (
 book_no INT PRIMARY KEY,
 book_name VARCHAR(100),
 author VARCHAR(100),
 quantity INT DEFAULT 0
);
-- Create Iss_rec Table (Issue Record)
CREATE TABLE Iss_rec (
 iss_no INT PRIMARY KEY,
 iss_date DATE,
 Mem_no INT,
 book_no INT,
 FOREIGN KEY (Mem_no) REFERENCES Membership(Mem_no),
 FOREIGN KEY (book_no) REFERENCES Book(book_no)
);
-- Insert data into Student table
INSERT INTO Student (Stud_no, Stud_name)
VALUES

(2, 'Jane Smith'),
(3, 'Alice Johnson');
-- Insert data into Membership table
INSERT INTO Membership (Mem_no, Stud_no)
VALUES

-- Insert data into Book table (with quantity)

(3, 'Data Structures and Algorithms', 'Mark Lee', 3);
-- Insert data into Iss_rec table
INSERT INTO Iss_rec (iss_no, iss_date, Mem_no, book_no)
VALUES

(2, '2025-04-02', 102, 1),
(3, '2025-04-03', 103, 2);
DELIMITER $$
CREATE TRIGGER UpdateBookQuantityAfterIssue
AFTER INSERT ON Iss_rec
FOR EACH ROW
BEGIN
 -- Decrease the quantity of the book by 1 after it is issued
 UPDATE Book
 SET quantity = quantity - 1
 WHERE book_no = NEW.book_no AND quantity > 0;
END $$
DELIMITER ;
-- Issue books (trigger will decrease quantity automatically)
